# Mini App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Telegram Mini App с чёрно-белым списком записей — просмотр, фильтры по категориям, inline-редактирование, удаление.

**Architecture:** Один HTML-файл в `backend/webapp/index.html`, отдаётся Express по `/app`. API-эндпоинты на том же сервере, аутентификация через Telegram initData HMAC-SHA256. Кнопка «📁 Записи» становится `web_app`-кнопкой.

**Tech Stack:** Node.js/Express, Vanilla JS (без фреймворков), Telegram Bot API Web Apps

```js
// backend/auth/telegram-webapp.js
import crypto from 'crypto'

export function verifyInitData(initData) {
  const params = new URLSearchParams(initData)
  const hash = params.get('hash')
  if (!hash) return null

  params.delete('hash')
  const dataCheckString = [...params.entries()]
    .sort(([a], [b]) => a.localeCompare(b))
    .map(([k, v]) => `${k}=${v}`)
    .join('\n')

  const secretKey = crypto.createHmac('sha256', 'WebAppData')
    .update(process.env.BOT_TOKEN)
    .digest()

  const expectedHash = crypto.createHmac('sha256', secretKey)
    .update(dataCheckString)
    .digest('hex')

  if (expectedHash !== hash) return null

  const userStr = params.get('user')
  if (!userStr) return null
  return JSON.parse(userStr)
}

export function webAppAuth(req, res, next) {
  const initData = req.headers['x-telegram-init-data']
  if (!initData) return res.status(401).json({ error: 'No initData' })
  const user = verifyInitData(initData)
  if (!user) return res.status(401).json({ error: 'Invalid initData' })
  req.telegramUser = user
  next()
}
```

```js
// Mini App API
app.get('/api/records', webAppAuth, async (req, res) => {
  try {
    const user = await getUser(String(req.telegramUser.id))
    if (!user?.encrypted_token) return res.json({ records: [] })
    const db = await getDriveForUser(user).readJson(user.encrypted_token, user.drive_file_id)
    res.json({ records: db.records || [] })
  } catch (e) {
    console.error('GET /api/records', e)
    res.status(500).json({ error: 'Failed to load' })
  }
})

app.patch('/api/records/:id', webAppAuth, async (req, res) => {
  try {
    const user = await getUser(String(req.telegramUser.id))
    if (!user?.encrypted_token) return res.status(404).json({ error: 'No storage' })
    const db = await getDriveForUser(user).readJson(user.encrypted_token, user.drive_file_id)
    const idx = db.records.findIndex(r => r.id === req.params.id)
    if (idx === -1) return res.status(404).json({ error: 'Not found' })
    const { data, comment } = req.body
    if (data !== undefined) db.records[idx].data = data
    if (comment !== undefined) db.records[idx].comment = comment
    await getDriveForUser(user).writeJson(user.encrypted_token, user.drive_file_id, db)
    res.json({ record: db.records[idx] })
  } catch (e) {
    console.error('PATCH /api/records', e)
    res.status(500).json({ error: 'Failed to save' })
  }
})

app.delete('/api/records/:id', webAppAuth, async (req, res) => {
  try {
    const user = await getUser(String(req.telegramUser.id))
    if (!user?.encrypted_token) return res.status(404).json({ error: 'No storage' })
    const db = await getDriveForUser(user).readJson(user.encrypted_token, user.drive_file_id)
    const before = db.records.length
    db.records = db.records.filter(r => r.id !== req.params.id)
    if (db.records.length === before) return res.status(404).json({ error: 'Not found' })
    await getDriveForUser(user).writeJson(user.encrypted_token, user.drive_file_id, db)
    res.json({ ok: true })
  } catch (e) {
    console.error('DELETE /api/records', e)
    res.status(500).json({ error: 'Failed to delete' })
  }
})
```

<script>
const tg = window.Telegram.WebApp
tg.ready()
tg.expand()

const BASE = ''
let allRecords = []
let activeFilter = 'все'
let openRowId = null

async function api(method, path, body) {
  const res = await fetch(BASE + path, {
    method,
    headers: {
      'Content-Type': 'application/json',
      'X-Telegram-Init-Data': tg.initData
    },
    body: body ? JSON.stringify(body) : undefined
  })
  if (!res.ok) throw new Error(await res.text())
  return res.json()
}

async function load() {
  try {
    const { records } = await api('GET', '/api/records')
    allRecords = records
    renderFilters()
    render()
  } catch (e) {
    document.getElementById('list').innerHTML = '<div class="state">Не удалось загрузить записи</div>'
  }
}

function getCategories() {
  const cats = [...new Set(allRecords.map(r => r.category).filter(Boolean))]
  return cats
}

function renderFilters() {
  const cats = getCategories()
  const filtersEl = document.getElementById('filters')
  filtersEl.innerHTML = ''
  const all = chip('все', activeFilter === 'все')
  filtersEl.appendChild(all)
  cats.forEach(c => filtersEl.appendChild(chip(c, activeFilter === c)))
}

function chip(label, active) {
  const el = document.createElement('div')
  el.className = 'chip' + (active ? ' active' : '')
  el.textContent = label
  el.onclick = () => { activeFilter = label; renderFilters(); render() }
  return el
}

function filtered() {
  if (activeFilter === 'все') return allRecords
  return allRecords.filter(r => r.category === activeFilter)
}

function render() {
  const records = filtered()
  const listEl = document.getElementById('list')
  const countEl = document.getElementById('count')
  countEl.textContent = allRecords.length + ' ' + plural(allRecords.length, 'запись', 'записи', 'записей')

  if (!records.length) {
    listEl.innerHTML = '<div class="state">Ещё ничего не сохранено</div>'
    return
  }

  listEl.innerHTML = ''
  records.forEach(rec => listEl.appendChild(makeRow(rec)))
}

function plural(n, one, few, many) {
  const mod10 = n % 10, mod100 = n % 100
  if (mod10 === 1 && mod100 !== 11) return one
  if (mod10 >= 2 && mod10 <= 4 && (mod100 < 10 || mod100 >= 20)) return few
  return many
}

function makeSub(rec) {
  const d = rec.data || {}
  const parts = [
    d.описание || d.услуга || d.специализация || d.навыки || d.тема || (rec.tags?.[0]),
    d.телефон || d.telegram || d.username || d.сайт || d.ссылка || d.источник_tg
  ].filter(Boolean)
  return parts.join(' · ')
}

function makeName(rec) {
  const d = rec.data || {}
  return d.имя || d.название || d.источник_имя || rec.category
}

function makeRow(rec) {
  const row = document.createElement('div')
  row.className = 'row' + (openRowId === rec.id ? ' open' : '')
  row.dataset.id = rec.id

  const name = makeName(rec)
  const sub = makeSub(rec)

  row.innerHTML = `
    <div class="row-main">
      <div class="row-text">
        <div class="row-name">${esc(name)}</div>
        ${sub ? `<div class="row-sub">${esc(sub)}</div>` : ''}
      </div>
      <div class="row-del">×</div>
    </div>
    <div class="edit-form">${makeForm(rec)}</div>
  `

  row.querySelector('.row-del').onclick = (e) => {
    e.stopPropagation()
    deleteRecord(rec.id, row)
  }
  row.querySelector('.row-main').onclick = () => toggleRow(rec.id)
  row.querySelector('.btn-save')?.addEventListener('click', (e) => {
    e.stopPropagation()
    saveRecord(rec, row)
  })

  return row
}

function makeForm(rec) {
  const d = rec.data || {}
  const fields = Object.entries(d).map(([k, v]) =>
    `<div class="field"><label>${esc(k)}</label><input data-key="${esc(k)}" value="${esc(String(v))}"></div>`
  ).join('')
  const commentField = `<div class="field comment"><label>комментарий</label><textarea data-comment>${esc(rec.comment || '')}</textarea></div>`
  return fields + commentField + `<button class="btn-save">Сохранить</button>`
}

function toggleRow(id) {
  openRowId = openRowId === id ? null : id
  render()
}

async function deleteRecord(id, row) {
  row.style.transition = 'opacity 0.25s'
  row.style.opacity = '0.3'
  try {
    await api('DELETE', `/api/records/${id}`)
    allRecords = allRecords.filter(r => r.id !== id)
    if (openRowId === id) openRowId = null
    renderFilters()
    render()
  } catch {
    row.style.opacity = '1'
  }
}

async function saveRecord(rec, row) {
  const btn = row.querySelector('.btn-save')
  btn.textContent = '...'
  btn.disabled = true
  const newData = {}
  row.querySelectorAll('input[data-key]').forEach(inp => {
    newData[inp.dataset.key] = inp.value
  })
  const comment = row.querySelector('textarea[data-comment]')?.value || ''
  try {
    const { record } = await api('PATCH', `/api/records/${rec.id}`, { data: newData, comment })
    const idx = allRecords.findIndex(r => r.id === rec.id)
    if (idx !== -1) allRecords[idx] = record
    openRowId = null
    renderFilters()
    render()
  } catch {
    btn.textContent = 'Сохранить'
    btn.disabled = false
  }
}

function esc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;')
}

load()
